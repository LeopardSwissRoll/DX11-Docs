Shader + HLSL
GPU에게 "어떻게 그려"

정점 데이터가 GPU에 도착했으면, "이걸 어떻게 처리할지" 알려줘야 한다. HLSL이라는 GPU 전용 언어로.

CPU ↔ GPU 사이의 약속 — InputLayout

GPU는 C++ 구조체를 모르니까 "이 바이트 덩어리를 어떻게 해석할지" 알려줘야 해.

CPU (FVertexColor)InputLayoutHLSL (VS_Input_Color)
_pos (FVector3D, 12B)POSITION, R32G32B32_FLOAT, offset=0float3 Pos : POSITION
_color (FVector4D, 16B)COLOR, R32G32B32A32_FLOAT, offset=12float4 Color : Color

Semantic = HLSL에서 데이터를 식별하는 이름. CPU 구조체 멤버와 HLSL semantic이 InputLayout을 통해 매칭됨.

AddInputLayoutDesc — 설명서 한 줄 추가

C++ InputLayout 요소 추가 Shader/GraphicShader.cpp
void GraphicShader::AddInputLayoutDesc(const char* semetic, uint32 sementicIndex,
    DXGI_FORMAT fmt, uint32 inputSlot, uint32 size,
    D3D11_INPUT_CLASSIFICATION inputSlotClass, uint32 instanceDataStepRate)
{
    D3D11_INPUT_ELEMENT_DESC desc = {};

    desc.SemanticName = semetic;                 // HLSL 시멘틱 이름
    desc.SemanticIndex = sementicIndex;          // 같은 이름 여러 개일 때 인덱스
    desc.Format = fmt;                           // 데이터 포맷
    desc.InputSlot = inputSlot;                  // 어떤 버텍스 버퍼에서 오는지
    desc.InputSlotClass = inputSlotClass;        // 정점 단위 / 인스턴스 단위
    desc.InstanceDataStepRate = instanceDataStepRate;
    desc.AlignedByteOffset = _inputLayoutSize;   // 정점 시작부터 이 데이터까지의 거리

    _inputLayoutSize += size;

    _descs.push_back(desc);
}

AlignedByteOffset 자동 계산 — POSITION 추가하면 offset=0, COLOR 추가하면 offset=12.

C++ GPU에 설명서 등록 Shader/GraphicShader.cpp
bool GraphicShader::CreateInputLayout()
{
    if (FAILED(Device::Instance().GetDevice()->CreateInputLayout(
        &_descs[0], (UINT)_descs.size(),
        _vsBlob->GetBufferPointer(), _vsBlob->GetBufferSize(),
        &_inputLayout)))
        return false;

    return true;
}
VS 바이트코드가 필요한 이유
GPU가 "이 InputLayout이 이 VS와 호환되는지" 검증하니까. 컴파일된 _vsBlob의 입력 시그니처와 대조함.

LoadVertexShader — HLSL 로드 + 컴파일

C++ VS 로드/컴파일 Shader/GraphicShader.cpp
bool GraphicShader::LoadVertexShader(const char* entryName, const char* fileName)
{
    auto hlslPath = DirectoryManager::Instance().GetPath("HLSL");
    if (!hlslPath.has_value())
        return false;

    std::filesystem::path fullPath;
    if (!DirectoryManager::Instance().GetFile(hlslPath.value(), fileName, OUT fullPath))
        return false;

    uint32 flag = 0;
#ifdef _DEBUG
    flag = D3DCOMPILE_DEBUG;
#endif

    ID3D10Blob* errorBlob = nullptr;

    if (FAILED(D3DCompileFromFile(fullPath.c_str(), nullptr, D3D_COMPILE_STANDARD_FILE_INCLUDE,
        entryName, "vs_5_0", flag, 0, _vsBlob.GetAddressOf(), &errorBlob)))
    {
#ifdef _DEBUG
        char errorText[255] = {};
        strcpy_s(errorText, (const char*)errorBlob->GetBufferPointer());
        LogManager::Instance().Debug(errorText);
#endif
        return false;
    }

    if (FAILED(Device::Instance().GetDevice()->CreateVertexShader(
        _vsBlob->GetBufferPointer(), _vsBlob->GetBufferSize(),
        nullptr, _vs.GetAddressOf())))
    {
        LogManager::Instance().Debug(__FILE__, __LINE__, "[CreateVertexShader Error]");
        return false;
    }

    return true;
}

struct VS_Input_Color
{
    float3 Pos : POSITION;     //CPU 버퍼에서 이 부분은 위치정보
    float4 Color : Color;      //CPU 버퍼에서 이 부분은 색상 정보
};

struct VS_Output_Color
{
    float4 Pos : SV_POSITION;   // 시스템 시멘틱 — 래스터화에 사용
    float4 Color : COLOR;       // 래스터라이저가 보간 → PS에 전달
};

Vertex Shader — 패스스루

HLSL HLSL/ColorMesh.fx
//버텍스 쉐이더
VS_Output_Color ColorMeshVS(VS_Input_Color input)
{
    VS_Output_Color output = (VS_Output_Color) 0;

    //output.Pos의 x, y, z에는 input.Pos의 x, y, z가 들어가고, w에는 1이 들어간다.
    output.Pos = float4(input.Pos, 1.f);
    output.Color = input.Color;

    return output;
}

현재 변환 없이 패스스루. 나중에 월드/뷰/프로젝션 행렬 곱이 여기 들어올 거야.

float4(input.Pos, 1.f) — 3D(x,y,z) → 4D 동차좌표(x,y,z,w). w=1은 "위치"를 의미.

Pixel Shader — 색 그대로 출력

//픽셀쉐이더
PS_Output_Single ColorMeshPS(VS_Output_Color input)
{
    PS_Output_Single output = (PS_Output_Single) 0;

VS에서 넘어온 Color를 SV_TARGET에 출력. 래스터라이저가 꼭짓점 사이를 보간해주니까 삼각형 내부가 그라데이션.

SetShader — 매 프레임 바인딩

C++ VS→PS→InputLayout 바인딩 Shader/GraphicShader.cpp
void GraphicShader::SetShader()
{
    Device::Instance().GetContext()->VSSetShader(_vs.Get(), nullptr, 0);
    Device::Instance().GetContext()->PSSetShader(_ps.Get(), nullptr, 0);

    Device::Instance().GetContext()->IASetInputLayout(_inputLayout.Get());
}

헤더에서 virtual void SetShader() final; — 자식 클래스가 재정의 불가.

ColorMeshShader::Init() — 전부 조립

C++ 4단계 조립 Shader/ColorMeshShader.cpp
bool ColorMeshShader::Init()
{
    if (!LoadVertexShader("ColorMeshVS", "ColorMesh.fx"))
        return false;

    if (!LoadPixelShader("ColorMeshPS", "ColorMesh.fx"))
        return false;

    AddInputLayoutDesc("POSITION", 0, DXGI_FORMAT_R32G32B32_FLOAT, 0, 12, D3D11_INPUT_PER_VERTEX_DATA, 0);
    AddInputLayoutDesc("COLOR", 0, DXGI_FORMAT_R32G32B32A32_FLOAT, 0, 16, D3D11_INPUT_PER_VERTEX_DATA, 0);

    if (!CreateInputLayout())
        return false;

    return true;
}
