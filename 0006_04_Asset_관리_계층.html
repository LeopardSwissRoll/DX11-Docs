AssetManager — 서브 매니저들의 매니저

C++ RegisterManager + Get Core/AssetManager.h
class AssetManager : public Singleton<AssetManager>
{
private:
    std::map<eAssetType, Ptr<class SubManager>> _managers;

public:
    bool Init();

    template<typename T>
    Ptr<T> Get(eAssetType type)
    {
        if (_managers.end() == _managers.find(type))
            return nullptr;

        return Cast<SubManager, T>(_managers[type]);
    }

    template<typename T>
    bool RegisterManager(eAssetType type)
    {
        Ptr<T> manager = New<T>();
        if (false == manager->Init())
        {
            Delete(manager);
            return false;
        }

        _managers[type] = manager;
        return true;
    }

    virtual void Destroy() override;
};
C++ Init — MeshManager 등록 Core/AssetManager.cpp
bool AssetManager::Init()
{
    if (!RegisterManager<MeshManager>(eAssetType::Mesh))
        return false;

    return true;
}
사용법
AssetManager::Instance().Get<MeshManager>(eAssetType::Mesh)->FindMesh("Triangle")

ShaderManager — 독립 싱글톤

C++ CreateShader + FindShader Shader/ShaderManager.h
class ShaderManager : public Singleton<ShaderManager>
{
protected:
    std::unordered_map<std::string, Ptr<Shader>> _shaders;

public:
    bool Init();
    virtual void Destroy() override;
    Ptr<Shader> FindShader(const std::string& name);

private:
    template<typename T>
    bool CreateShader(const std::string& name)
    {
        Ptr<Shader> shader = FindShader(name);
        if (shader) return false;

        shader = New<T>();
        if (false == shader->Init())
        {
            Delete(shader);
            return false;
        }
        _shaders[name] = shader;
    }
};
C++ Init Shader/ShaderManager.cpp
bool ShaderManager::Init()
{
    CreateShader<ColorMeshShader>("ColorMeshShader");
    return true;
}

Destroy 순서 — Init의 역순

C++ Core/GameEngine.cpp
void GameEngine::Destroy()
{
    LogManager::Instance().Destroy();
    DirectoryManager::Instance().Destroy();

    AssetManager::Instance().Destroy();
    ShaderManager::Instance().Destroy();

#ifdef _USE_MEMORY_POOL
    MemoryPool::Instance().Destroy();
#elif defined(_USE_OBJECT_POOL)
    ObjectPool::Instance().Destroy()
#endif
}

GPU 리소스(Asset, Shader)를 먼저 해제하고, 메모리 풀을 나중에 정리.
